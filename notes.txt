   $ sudo mkdir /opt/wiremock-1.48
   $ sudo chown pi:pi /opt/wiremock-1.48
   $ sudo ln -s wiremock-1.48 /opt/wiremock
   $ cd /opt/wiremock
   $ wget 'http://repo1.maven.org/maven2/com/github/tomakehurst/wiremock/1.48/wiremock-1.48-standalone.jar'
   $ chmod 444 wiremock-1.48-standalone.jar
